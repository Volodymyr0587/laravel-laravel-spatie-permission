`git clone https://github.com/Volodymyr0587/laravel-spatie-permission`

`cd laravel-spatie-permission`

`composer install`

`cp .env.example .env`

`php artisan key:generate`

`create mysql database laravel_spatie_permission`

`php artisan migrate`

`php artisan storage:link`

`php artisan db:seed`

`npm install`

`npm run dev`

`php artisan serve`
